#include<stdio.h>

int main () {
    
    int d, i, k, loc=1,max_num ;
    printf ("total number of input:");
    scanf ("%d", &d) ;
    
    int D[d] ;
    for (i=0; i <d; i++) {
    printf ("enter number:%d",i++) ;
    scanf ("%d", &D[i]);
    }
    
    max_num = D[0] ;
    for (k=0; k<d; k++) {
        if (max_num <D[k]) {
            max_num = D[k] ;
            loc = k+1 ;
        }
    }
    
    printf ("highest number:%d\n", max_num) ;
    printf ("Location: %d\n", loc) ;
    
    return 0;
}
